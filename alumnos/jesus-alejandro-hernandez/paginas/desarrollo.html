Programación Arduino

 <const int pinBoton = 4;   // Botón a GND>
 <const int pinBuzzer = 7;  // Buzzer pasivo o activo>

void setup() {
  pinMode(pinBoton, INPUT_PULLUP); // Activa resistencia interna
  pinMode(pinBuzzer, OUTPUT);
  digitalWrite(pinBuzzer, LOW);
}

void loop() {
  if (digitalRead(pinBoton) == LOW) { // Botón presionado
    tone(pinBuzzer, 2000);  // Suena a 2kHz
    delay(200);             // Dura 200ms
    noTone(pinBuzzer);      // Se apaga
    delay(200);             // Antirebote simple
  }
}
