const int boton = 2;
const int buzzer = 8;
const int led = 13;

void setup() {
  pinMode(boton, INPUT_PULLUP);
  pinMode(buzzer, OUTPUT);
  pinMode(led, OUTPUT);
}

void loop() {
  if(digitalRead(boton) == LOW){
    digitalWrite(led, HIGH);
    tone(buzzer, 1000);
    delay(1000);
    noTone(buzzer);
    digitalWrite(led, LOW);
  }
}
